                                                                                                                                                                                                                                                                                                                                                            # 配置项

                                                                                                                                                                                                                                                                                                                                                            openGemini 配置项解释

                                                                                                                                                                                                                                                                                                                                                            # [common]

                                                                                                                                                                                                                                                                                                                                                            common是ts-sql、ts-store、ts-meta公共的配置项。

                                                                                                                                                                                                                                                                                                                                                            # meta-join 必填

                                                                                                                                                                                                                                                                                                                                                            • 类型: []string

                                                                                                                                                                                                                                                                                                                                                            • 默认值: 无

                                                                                                                                                                                                                                                                                                                                                            • SQL / STOR和META的RPC通信地址。比如：["127.0.0.1:8092", "127.0.0.2:8092", "127.0.0.3:8092"]

                                                                                                                                                                                                                                                                                                                                                            # ha-enable

                                                                                                                                                                                                                                                                                                                                                            • 类型: bool

                                                                                                                                                                                                                                                                                                                                                            • 默认值: false

                                                                                                                                                                                                                                                                                                                                                            • 基于共享存储的HA开关，目前仅支持共享存储可设置为true。

                                                                                                                                                                                                                                                                                                                                                            • 生产环境不建议开启。

                                                                                                                                                                                                                                                                                                                                                            # executor-memory-size-limit

                                                                                                                                                                                                                                                                                                                                                            • 类型: string | toml.Size
